    <script>
        // Elementos del DOM
        const backgroundImage = document.getElementById('backgroundImage');
        const imageUpload = document.getElementById('imageUpload');
        const title = document.getElementById('title');
        const subtitle = document.getElementById('subtitle');
        const whatsappButton = document.getElementById('whatsappButton');
        const fontButton = document.getElementById('fontButton');
        const colorButton = document.getElementById('colorButton');
        const fontSelector = document.getElementById('fontSelector');
        const colorSelector = document.getElementById('colorSelector');
        
        // Cargar datos desde localStorage
        window.addEventListener('load', function() {
            // Cargar imagen
            const savedImage = localStorage.getItem('backgroundImage');
            if (savedImage) {
                backgroundImage.style.backgroundImage = `url(${savedImage})`;
            }
            
            // Cargar fuente
            const savedFont = localStorage.getItem('fontFamily');
            if (savedFont) {
                title.style.fontFamily = savedFont;
                subtitle.style.fontFamily = savedFont;
                whatsappButton.style.fontFamily = savedFont;
            }
            
            // Cargar color
            const savedColor = localStorage.getItem('textColor');
            if (savedColor) {
                title.style.color = savedColor;
            }
        });
        
        // Cambiar imagen
        imageUpload.addEventListener('change', function(e) {
            if (e.target.files && e.target.files[0]) {
                const reader = new FileReader();
                reader.onload = function(event) {
                    const imageUrl = event.target.result;
                    backgroundImage.style.backgroundImage = `url(${imageUrl})`;
                    localStorage.setItem('backgroundImage', imageUrl);
                }
                reader.readAsDataURL(e.target.files[0]);
            }
        });
        
        // Mostrar/Ocultar selector de fuentes
        fontButton.addEventListener('click', function() {
            fontSelector.classList.toggle('show');
            colorSelector.classList.remove('show');
        });
        
        // Mostrar/Ocultar selector de colores
        colorButton.addEventListener('click', function() {
            colorSelector.classList.toggle('show');
            fontSelector.classList.remove('show');
        });
        
        // Seleccionar fuente
        document.querySelectorAll('.font-option').forEach(option => {
            option.addEventListener('click', function() {
                const font = this.getAttribute('data-font');
                title.style.fontFamily = font;
                subtitle.style.fontFamily = font;
                whatsappButton.style.fontFamily = font;
                localStorage.setItem('fontFamily', font);
                fontSelector.classList.remove('show');
            });
        });
        
        // Seleccionar color
        document.querySelectorAll('.color-option').forEach(option => {
            option.addEventListener('click', function() {
                const color = this.getAttribute('data-color');
                title.style.color = color;
                localStorage.setItem('textColor', color);
                colorSelector.classList.remove('show');
            });
        });
        
        // Cerrar selectores al hacer clic fuera
        document.addEventListener('click', function(e) {
            if (!e.target.closest('.controls') && !e.target.closest('.font-selector') && !e.target.closest('.color-selector')) {
                fontSelector.classList.remove('show');
                colorSelector.classList.remove('show');
            }
        });
        
        // Bot√≥n de WhatsApp
        whatsappButton.addEventListener('click', function() {
            const phoneNumber = '1234567890';
            const message = 'Hola, quiero mi usuario gratis para Ases del Norte';
            window.open(`https://wa.me/${phoneNumber}?text=${encodeURIComponent(message)}`, '_blank');
        });
